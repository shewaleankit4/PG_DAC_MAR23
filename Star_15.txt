class Star_15
	{

public static void main(String args[]) 
		{
			for(i=1;i<=4;i++)
			{
				for(int j=1;j<=4;j++)
				{
					if(j==1)
					System.out.print("*");
					else if(i==j)
					System.out.print("*");
					else if(i==4)
					System.out.print("*");
					else
					System.out.print(" ");
				}
				System.out.println();
			}
		}
	}
